                  # 3. HomeBrew

                  踩坑：推荐中科院的源

                  1、安装地址：https://zhuanlan.zhihu.com/p/111014448 (opens new window)

                  2、换源地址：https://www.zhihu.com/question/31360766/answer/749386652 (opens new window)

                  # 替换 Homebrew
git -C "$(brew --repo)" remote set-url origin https://mirrors.ustc.edu.cn/brew.git

# 替换 Homebrew Core
git -C "$(brew --repo homebrew/core)" remote set-url origin https://mirrors.ustc.edu.cn/homebrew-core.git

# 替换 Homebrew Cask
git -C "$(brew --repo homebrew/cask)" remote set-url origin https://mirrors.ustc.edu.cn/homebrew-cask.git

# 替换 Homebrew-bottles
# 对于 bash 用户：
echo 'export HOMEBREW_BOTTLE_DOMAIN=https://mirrors.ustc.edu.cn/homebrew-bottles' >> ~/.bash_profile
source ~/.bash_profile

# 对于 zsh 用户：
echo 'export HOMEBREW_BOTTLE_DOMAIN=https://mirrors.ustc.edu.cn/homebrew-bottles' >> ~/.zshrc
source ~/.zshrc

                  # 4. Nginx

                  install: brew install nginx

                  nginx 路径: /usr/local/etc/nginx/

                  sudo nginx             # 启动
sudo nginx -s reload   # 重启
sudo nginx -s stop     # 结束

                  # 5. 终端配置

                  tree

                  命令行打印目录树

                  # 安装
brew install tree

# 使用
# 我们要在目录遍历时使用 -L 参数指定遍历层级
tree -L 2 # 遍历两级目录

                  autojump

                  快捷目录跳转

                  # 安装
brew install autojump

# 在 .zshrc 中添加插件
vim ~/.zshrc # 

# 找到 plugins 并添加 autojump插件
plugins = ( git autojump )

# 重启 zsh
source ~/.zshrc
